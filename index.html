function doPost(e) {
  var data = JSON.parse(e.postData.contents);
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var action = data.action || 'attendance';

  if (action === 'attendance') {
    var sheet = ss.getSheetByName('Attendance');
    if (!sheet) {
      sheet = ss.insertSheet('Attendance');
      sheet.appendRow(['Date','Time','Chinese Name','English Name']);
      sheet.getRange('1:1').setFontWeight('bold');
    }
    sheet.appendRow([data.date, data.time, data.chinese||'', data.english||'']);
    return ok({status:'ok'});
  }

  if (action === 'initMasterList') {
    var ml = ss.getSheetByName('Master List');
    if (!ml) ml = ss.insertSheet('Master List');
    ml.clear();
    ml.appendRow(['Chinese Name','English Name','Service']);
    ml.getRange('1:1').setFontWeight('bold');
    var src = ss.getSheetByName('Form Responses 1')
           || ss.getSheets()[0];
    var rows = src.getDataRange().getValues();
    var seen = {};
    for (var i = 1; i < rows.length; i++) {
      var cn = (rows[i][1]||'').toString().trim();
      var en = (rows[i][2]||'').toString().trim();
      var svc = (rows[i][5]||'').toString().trim();
      if (!cn && !en) continue;
      var key = cn + '|' + en;
      seen[key] = [cn, en, svc];
    }
    var keys = Object.keys(seen).sort();
    for (var k = 0; k < keys.length; k++) {
      ml.appendRow(seen[keys[k]]);
    }
    return ok({status:'ok', count: keys.length});
  }

  if (action === 'addName') {
    var ml = getOrCreateML(ss);
    ml.appendRow([data.chinese||'', data.english||'', data.service||'']);
    return ok({status:'ok'});
  }

  if (action === 'updateName') {
    var ml = getOrCreateML(ss);
    var row = findRow(ml, data.oldChinese, data.oldEnglish);
    if (row > 0) {
      ml.getRange(row,1).setValue(data.chinese||'');
      ml.getRange(row,2).setValue(data.english||'');
    }
    return ok({status: row > 0 ? 'ok' : 'not_found'});
  }

  if (action === 'deleteName') {
    var ml = getOrCreateML(ss);
    var row = findRow(ml, data.chinese, data.english);
    if (row > 0) ml.deleteRow(row);
    return ok({status: row > 0 ? 'ok' : 'not_found'});
  }

  return ok({status:'unknown_action'});
}

function getOrCreateML(ss) {
  var ml = ss.getSheetByName('Master List');
  if (!ml) {
    ml = ss.insertSheet('Master List');
    ml.appendRow(['Chinese Name','English Name','Service']);
    ml.getRange('1:1').setFontWeight('bold');
  }
  return ml;
}

function findRow(sheet, cn, en) {
  var data = sheet.getDataRange().getValues();
  for (var i = 1; i < data.length; i++) {
    if (data[i][0] == cn && data[i][1] == en) return i+1;
  }
  return -1;
}

function ok(obj) {
  return ContentService.createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}

function doGet(e) {
  return ContentService.createTextOutput('Script is running.');
}
