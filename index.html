function doPost(e) {
  var data = JSON.parse(e.postData.contents);
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var action = data.action || 'attendance';

  if (action === 'attendance') {
    var sheet = ss.getSheetByName('Attendance');
    if (!sheet) {
      sheet = ss.insertSheet('Attendance');
      sheet.appendRow(['Date','Time','Chinese Name','English Name','Service','IsKid','IsNew','DeviceId','PersonId']);
      sheet.getRange('1:1').setFontWeight('bold');
    }
    sheet.appendRow([data.date, data.time, data.chinese||'', data.english||'',
      data.service||'', data.isKid?'Y':'', data.isNew?'Y':'',
      data.deviceId||'', data.personId||'']);
    return ok({status:'ok'});
  }

  if (action === 'audit') {
    var sheet = ss.getSheetByName('Audit');
    if (!sheet) {
      sheet = ss.insertSheet('Audit');
      sheet.appendRow(['Timestamp','DeviceId','DeviceName','Action','Details']);
      sheet.getRange('1:1').setFontWeight('bold');
    }
    sheet.appendRow([data.timestamp, data.deviceId||'', data.deviceName||'',
      data.auditAction||'', data.details||'']);
    return ok({status:'ok'});
  }

  if (action === 'heartbeat') {
    var sheet = ss.getSheetByName('Devices');
    if (!sheet) {
      sheet = ss.insertSheet('Devices');
      sheet.appendRow(['DeviceId','DeviceName','UserAgent','LastSeen','FirstSeen']);
      sheet.getRange('1:1').setFontWeight('bold');
    }
    var rows = sheet.getDataRange().getValues();
    var found = false;
    for (var i = 1; i < rows.length; i++) {
      if (rows[i][0] === data.deviceId) {
        sheet.getRange(i+1,2).setValue(data.deviceName||'');
        sheet.getRange(i+1,3).setValue(data.userAgent||'');
        sheet.getRange(i+1,4).setValue(data.timestamp);
        found = true;
        break;
      }
    }
    if (!found) {
      sheet.appendRow([data.deviceId, data.deviceName||'',
        data.userAgent||'', data.timestamp, data.timestamp]);
    }
    return ok({status:'ok'});
  }

  if (action === 'initMasterList') {
    var ml = ss.getSheetByName('Master List');
    if (!ml) ml = ss.insertSheet('Master List');
    ml.clear();
    ml.appendRow(['Chinese Name','English Name','Service','Group']);
    ml.getRange('1:1').setFontWeight('bold');
    var src = ss.getSheetByName('Form Responses 1')
           || ss.getSheets()[0];
    var rows = src.getDataRange().getValues();
    var seen = {};
    for (var i = 1; i < rows.length; i++) {
      var cn = (rows[i][1]||'').toString().trim();
      var en = (rows[i][2]||'').toString().trim();
      var svc = (rows[i][5]||'').toString().trim();
      if (!cn && !en) continue;
      var key = cn + '|' + en;
      seen[key] = [cn, en, svc, ''];
    }
    // Merge extra names if provided
    if (data.extraNames && data.extraNames.length) {
      for (var j = 0; j < data.extraNames.length; j++) {
        var e = data.extraNames[j];
        var key2 = (e.chinese||'') + '|' + (e.english||'');
        if (!seen[key2]) {
          seen[key2] = [e.chinese||'', e.english||'',
            e.service||'', e.group||''];
        } else if (e.group) {
          seen[key2][3] = e.group;
        }
      }
    }
    // Apply family groupings if provided
    if (data.groups) {
      for (var gn in data.groups) {
        var members = data.groups[gn];
        for (var m = 0; m < members.length; m++) {
          var gk = (members[m][0]||'')+'|'+(members[m][1]||'');
          if (seen[gk]) seen[gk][3] = gn;
        }
      }
    }
    var keys = Object.keys(seen).sort();
    for (var k = 0; k < keys.length; k++) {
      ml.appendRow(seen[keys[k]]);
    }
    return ok({status:'ok', count: keys.length});
  }

  if (action === 'addName') {
    var ml = getOrCreateML(ss);
    ml.appendRow([data.chinese||'', data.english||'',
      data.service||'', data.group||'']);
    return ok({status:'ok'});
  }

  if (action === 'updateName') {
    var ml = getOrCreateML(ss);
    var row = findRow(ml, data.oldChinese, data.oldEnglish);
    if (row > 0) {
      ml.getRange(row,1).setValue(data.chinese||'');
      ml.getRange(row,2).setValue(data.english||'');
      if (data.group !== undefined)
        ml.getRange(row,4).setValue(data.group||'');
    }
    return ok({status: row > 0 ? 'ok' : 'not_found'});
  }

  if (action === 'deleteName') {
    var ml = getOrCreateML(ss);
    var row = findRow(ml, data.chinese, data.english);
    if (row > 0) ml.deleteRow(row);
    return ok({status: row > 0 ? 'ok' : 'not_found'});
  }

  if (action === 'bulkAdd') {
    var ml = getOrCreateML(ss);
    var existing = {};
    var allData = ml.getDataRange().getValues();
    for (var i = 1; i < allData.length; i++) {
      existing[allData[i][0]+'|'+allData[i][1]] = i+1;
    }
    var added = 0;
    var items = data.names || [];
    for (var j = 0; j < items.length; j++) {
      var n = items[j];
      var nk = (n.chinese||'')+'|'+(n.english||'');
      if (!existing[nk]) {
        ml.appendRow([n.chinese||'',n.english||'',
          n.service||'',n.group||'']);
        added++;
      } else if (n.group) {
        var r = existing[nk];
        ml.getRange(r,4).setValue(n.group);
      }
    }
    return ok({status:'ok', added: added});
  }

  return ok({status:'unknown_action'});
}

function getOrCreateML(ss) {
  var ml = ss.getSheetByName('Master List');
  if (!ml) {
    ml = ss.insertSheet('Master List');
    ml.appendRow(['Chinese Name','English Name','Service','Group']);
    ml.getRange('1:1').setFontWeight('bold');
  }
  return ml;
}

function findRow(sheet, cn, en) {
  var data = sheet.getDataRange().getValues();
  for (var i = 1; i < data.length; i++) {
    if (data[i][0] == cn && data[i][1] == en) return i+1;
  }
  return -1;
}

function ok(obj) {
  return ContentService.createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}

function doGet(e) {
  return ContentService.createTextOutput('Thrive Name Tags API v2.0');
}
